76static int ft_get_tkns(char *line, t_token **tks)
77{
78 char *tmp;
79 t_tk_ops tk;
80
81 tmp = line;
82 while (line && *line)
83 {
84 tk = ft_get_tk(line);
85 if ((tk.tkn != NO_TOKEN) && (tmp != line))
86 ft_tk_add(tks, ft_tk_new(tmp, TK_CMD, (line - tmp)));
87 if (tk.tkn != NO_TOKEN)
88 {
89 line += tk.len;
90 if (tk.type != TK_BLANK)
91 ft_tk_add(tks, ft_tk_new(tk.tkn, tk.type, tk.len));
92 tmp = line;
93 }
94 else if (((*line == '\'') || (*line == '\"')) && ft_has_match(&line))
95 return (FAILURE);
96 else
97 ++line;
98 }
99 if (tmp != line)
100 ft_tk_add(tks, ft_tk_new(tmp, TK_CMD, (line - tmp)));
101 ft_free(tk.tkn);
102 return (SUCCESS);
103}

113static t_tk_ops ft_get_tk(char *tk)
114{
115 t_tk_ops ops[16];
116 t_tk_ops ret;
117 int i;
118
119 ft_init_ops(ops);
120 ret = (t_tk_ops){0, TK_CMD, 0};
121 i = 0;
122 if (!ft_isspace(tk[0]))
123 {
124 while (tk[i] && (!ft_isspace(tk[i])))
125 ++i;
126 ret = (t_tk_ops){ft_substr(tk, 0, i), TK_CMD, i};
127 }
128 else
129 ret = (t_tk_ops){ft_substr(tk, 0, 1), TK_BLANK, 1};
130 i = -1;
131 while (ops[++i].tkn)
132 if (!ft_strncmp(tk, ops[i].tkn, ops[i].len))
133 return (ops[i]);
134 return (ret);
135}
