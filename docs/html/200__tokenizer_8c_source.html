81static int ft_get_tkns(char *line, t_token **tks)
82{
83 char *tmp;
84 t_tk_ops tk;
85
86 tmp = line;
87 while (line && *line)
88 {
89 tk = ft_get_tk(line);
90 if ((tk.tkn != NO_TOKEN) && (tmp != line))
91 ft_tk_add(tks, ft_tk_new(tmp, TK_CMD, (line - tmp)));
92 if (tk.tkn != NO_TOKEN)
93 {
94 line += tk.len;
95 if (tk.type != TK_BLANK)
96 ft_tk_add(tks, ft_tk_new(tk.tkn, tk.type, tk.len));
97 tmp = line;
98 }
99 else if (((*line == '\'') || (*line == '\"')) && ft_has_match(&line))
100 return (FAILURE);
101 else
102 ++line;
103 }
104 if (tmp != line)
105 ft_tk_add(tks, ft_tk_new(tmp, TK_CMD, (line - tmp)));
106 ft_free(tk.tkn);
107 return (SUCCESS);
108}

121static t_tk_ops ft_get_tk(char *tk)
122{
123 t_tk_ops ops[16];
124 t_tk_ops ret;
125 int i;
126
127 ft_init_ops(ops);
128 ret = (t_tk_ops){0, TK_CMD, 0};
129 i = 0;
130 if (!ft_isspace(tk[0]))
131 {
132 while (tk[i] && (!ft_isspace(tk[i])))
133 ++i;
134 ret = (t_tk_ops){ft_substr(tk, 0, i), TK_CMD, i};
135 }
136 else
137 ret = (t_tk_ops){ft_substr(tk, 0, 1), TK_BLANK, 1};
138 i = -1;
139 while (ops[++i].tkn)
140 if (!ft_strncmp(tk, ops[i].tkn, ops[i].len))
141 return (ops[i]);
142 return (ret);
143}
