13/* ft_rgb : Create a color quick and dirty from RGB values
14 * (w/out accounting for endianness)
15 *
16 * (red << 16) : Shifts the bits of the red value to the left by 16 bits
17 * This places the red value in the most significant bits of the
18 * color value to be returned;
19 * (green << 8) : Shifts the bits of the green value to the left by 8 bits
20 * This places the green value in the middle bits of the color
21 * value to be returned:
22 * (red << 16 | green << 8 | blue) : Combines the shifted values of red, green,
23 * and blue. this operation concatenated the bits of the shiftend
24 * values with the bits of blue;
25 * */
26int ft_argb(int alpha, int red, int green, int blue)
27{
28 return (alpha << 24 | red << 16 | green << 8 | blue);
29}
