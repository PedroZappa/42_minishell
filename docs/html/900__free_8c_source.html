93int ft_free_cmds(t_cmd *cmds, int n_cmds)
94{
95 int i;
96 int j;

98 if (!cmds)
99 return (FAILURE);
100 i = -1;
101 while (++i < n_cmds)
102 {
103 j = -1;
104 while (++j < cmds[i].argc)
105 if (cmds[i].argv[j])
106 ft_free(cmds[i].argv[j]);
107 free(cmds[i].argv);
108 if (cmds[i].in.name)
109 ft_free(cmds[i].in.name);
110 if (cmds[i].out.name)
111 ft_free(cmds[i].out.name);
112 if (cmds[i].in.heredoc)
113 ft_lstclear(&cmds[i].in.heredoc, free);
114 }
115 if (cmds)
116 free(cmds);
117 return (SUCCESS);
118}
