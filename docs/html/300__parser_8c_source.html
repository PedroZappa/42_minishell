46int ft_parser(t_shell *sh, char **line_buf)
47{
48 t_token *tks;
49
50 tks = NULL;
51 ft_readline(&line_buf);
52 if (ft_tokenizer(sh, line_buf, &tks))
53 return (ft_free_tks(&tks), FAILURE);
54 if (ft_check_syntax(tks))
55 return (ft_free_tks(&tks), FAILURE);
56 sh->n_cmds = ft_count_cmds(tks);
57 sh->cmds = ft_calloc(sh->n_cmds, sizeof(t_cmd));
58 if (!sh->cmds)
59 return (ft_err(MALLOC_ERR, errno), FAILURE);
60 if (sh->n_cmds == NO_CMDS)
61 return (ft_free_nocmds(sh->cmds, sh->n_cmds, &tks));
62 ft_count_args(sh, tks);
63 if (ft_parse_cmds(tks, sh->cmds, 0, 0))
64 return (ft_free_tks(&tks), FAILURE);
65 return (ft_free_tks(&tks), SUCCESS);
66}

80static int ft_check_syntax(t_token *tk)
81{
82 if (tk && (tk->type == TK_PIPE || tk->type == TK_OR))
83 return (ft_syntax_err(tk->name, FAILURE));
84 while (tk)

178static int ft_parse_cmds(t_token *tks, t_cmd *cmds, int i, int j)
179{
180 while (tks)
181 {
182 cmds[i].argv = ft_calloc((cmds[i].argc + 1), sizeof(char *));
183 if (!cmds[i].argv)
184 return (ft_err(MALLOC_ERR, errno), FAILURE);
185 j = 0;
186 while (tks && (tks->type != TK_PIPE) && (tks->type != TK_OR))
187 {
188 if (tks->type == TK_CMD)
189 cmds[i].argv[j++] = ft_strdup(tks->name);
190 if (tks)
191 tks = tks->next;
192 }
193 if (tks)
194 tks = tks->next;
195 cmds[i].argv[j] = NULL;
196 ++i;
197 }
198 return (SUCCESS);
199}
